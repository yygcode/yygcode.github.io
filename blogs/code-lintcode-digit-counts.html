Code - Lintcode Digit Counts问题成绩排名第一

领扣刷题，Digit Counts成绩排名第一，记录下。具体算法说明后补。 

class Solution {
public:
    /**
     * @param k: An integer
     * @param n: An integer
     * @return: An integer denote the count of digit k in 1..n
     */
    int digitCounts(int k, int n) {
        // write your code here
        int xd, xbase, dbase = 1, div = 10, c = 0, k0penalty = 0;

        if (n%10 >= k)
                c += 1;

        while(div <= n) {
                xd = n%(div*10);
                xd /= div;
                xbase = div/10*dbase;

                c += xd*xbase;
                if (xd > k)
                        c += div;
                else if (xd == k)
                        c += (n%div) + 1;

                k0penalty += div;
                div *= 10;
                ++dbase;
        }

        if (k == 0)
                c -= k0penalty;

        return c;
    }
};

#include <stdio.h>

int countk(int n, int k)
{
        int i, v, c = 0;

        for (i = 0; i <= n; ++i) {
                v = i;
                do {
                        if (v%10 == k)
                                ++c;
                        v /= 10;
                } while (v);
        }

        return c;
}
